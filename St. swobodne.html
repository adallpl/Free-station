double Wspolrzedne(double Delta_Wsp_prz, double Wspol_Poprawka)
{
    double wynik;
    wynik=Delta_Wsp_prz+Wspol_Poprawka;
    return wynik;
}
double Azymut(double del_Y, double del_X)
{
    double wynik;
    wynik = atan2(del_Y, del_X)*(200/M_PI);
    if(wynik<0)
    {
        wynik+=400.0;
    }
    return wynik;
}
